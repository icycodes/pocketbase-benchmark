package main

import (
	"fmt"
	"log"

	"github.com/pocketbase/pocketbase"
	"github.com/pocketbase/pocketbase/core"
)

func main() {
	app := pocketbase.New()
	app.OnServe().BindFunc(func(e *core.ServeEvent) error {
		categories, err := e.App.FindCollectionByNameOrId("categories")
		if err != nil {
			return err
		}
		if err := e.App.Delete(categories); err != nil {
			return err
		}
		fmt.Println("Deleted categories!")
		return nil
	})
	if err := app.Start(); err != nil {
		log.Fatal(err)
	}
}
