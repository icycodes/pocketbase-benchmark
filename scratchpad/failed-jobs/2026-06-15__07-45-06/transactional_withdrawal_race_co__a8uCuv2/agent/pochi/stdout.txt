package main

import (
    "log"

    "github.com/pocketbase/pocketbase"
    "github.com/pocketbase/pocketbase/core"
    "github.com/pocketbase/pocketbase/apis"
)

func main() {
    app := pocketbase.New()

    app.OnRecordCreateRequest("withdrawals").BindFunc(func(e *core.RecordRequestEvent) error {
        return e.App.RunInTransaction(func(txApp core.App) error {
            walletId := e.Record.GetString("wallet_id")
            amount := e.Record.GetFloat("amount")

            wallet, err := txApp.FindRecordById("wallets", walletId)
            if err != nil {
                return apis.NewBadRequestError("Wallet not found", nil)
            }

            balance := wallet.GetFloat("balance")
            if balance < amount {
                return apis.NewBadRequestError("Insufficient funds", nil)
            }

            wallet.Set("balance", balance - amount)
            if err := txApp.Save(wallet); err != nil {
                return err
            }

            return e.Next()
        })
    })

    if err := app.Start(); err != nil {
        log.Fatal(err)
    }
}

    "github.com/pocketbase/pocketbase/core"
    "github.com/pocketbase/pocketbase/apis"
)

func main() {
    app := pocketbase.New()

    app.OnBootstrap().BindFunc(func(e *core.BootstrapEvent) error {
        if err := e.Next(); err != nil {
            return err
        }

        // Create wallets collection if not exists
        wallets, err := e.App.FindCollectionByNameOrId("wallets")
        if err != nil {
            wallets = core.NewCollection(&core.Collection{
                Name: "wallets",
                Type: core.CollectionTypeBase,
                Fields: core.NewFieldList(
                    &core.NumberField{
                        Name: "balance",
                    },
                ),
            })
            e.App.Save(wallets)
        }

        // Create withdrawals collection if not exists
        withdrawals, err := e.App.FindCollectionByNameOrId("withdrawals")
        if err != nil {
            withdrawals = core.NewCollection(&core.Collection{
                Name: "withdrawals",
                Type: core.CollectionTypeBase,
                Fields: core.NewFieldList(

                        Name: "wallet_id",
                        CollectionId: wallets.Id,
                        MaxSelect: 1,
                    },
                    &core.NumberField{
                        Name: "amount",
                    },
                ),
            })
            e.App.Save(withdrawals)
        }

        return nil
    })

        return e.App.RunInTransaction(func(txApp core.App) error {
