def sse_client(token, label, result_queue, stop_event):
    """Use raw socket to properly handle SSE with PocketBase realtime API"""
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(5)
        sock.connect(("localhost", 8090))
        
        # Step 1: GET /api/realtime to establish SSE connection
        request = f"GET /api/realtime HTTP/1.1\r\nHost: localhost:8090\r\nConnection: keep-alive\r\nAccept: text/event-stream\r\n"
        if token:
            request += f"Authorization: {token}\r\n"
        request += "\r\n"
        sock.sendall(request.encode())
        
        # Read the response headers
        response = b""
        while b"\r\n\r\n" not in response:
            chunk = sock.recv(4096)
            if not chunk:
                break
            response += chunk
        
        print(f"  [{label}] Connected to SSE, response headers received")
        
        # Step 2: POST /api/realtime to subscribe (on the same connection via HTTP pipelining)
        sub_body = json.dumps({"subscriptions": ["notifications"]})
        sub_request = f"POST /api/realtime HTTP/1.1\r\nHost: localhost:8090\r\nContent-Type: application/json\r\nContent-Length: {len(sub_body)}\r\n"
        if token:
            sub_request += f"Authorization: {token}\r\n"
        sub_request += f"\r\n{sub_body}"
        sock.sendall(sub_request.encode())
        
        # Read response for the POST
        post_response = b""
        while b"\r\n\r\n" not in post_response:
            chunk = sock.recv(4096)
            if not chunk:
                break
            post_response += chunk
        
        print(f"  [{label}] Subscription sent, response: {post_response[:200]}")
        
        # Now listen for SSE events
        buffer = b""
        start_time = time.time()
        while not stop_event.is_set() and (time.time() - start_time) < 8:
            try:
                sock.settimeout(0.5)
                chunk = sock.recv(65536)
                if not chunk:
                    break
                buffer += chunk
                
                # Parse SSE events
                while b"\n\n" in buffer:
                    event_str, buffer = buffer.split(b"\n\n", 1)
                    text = event_str.decode("utf-8", errors="replace")
                    for line in text.split("\n"):
                        if line.startswith("data: "):
                            try:
                                data = json.loads(line[6:])
                                result_queue.put(data)
                                print(f"  [{label}] SSE event: {json.dumps(data)[:200]}")
                            except Exception as e:
                                print(f"  [{label}] Parse error: {e} for line: {line[:100]}")
            except socket.timeout:
                continue
            except Exception as e:
                print(f"  [{label}] Read error: {e}")
                break
        
        sock.close()
    except Exception as e:
        print(f"  [{label}] Connection error: {e}")
        import traceback
        traceback.print_exc()

def curl(method, url, headers=None, data=None, timeout=30):
    cmd = ["curl", "-s", "--no-buffer", "-X", method, url]
    if headers:
        for k, v in headers.items():
            cmd += ["-H", f"{k}: {v}"]
    if data:
        cmd += ["-H", "Content-Type: application/json", "-d", json.dumps(data)]
    out = subprocess.check_output(cmd, timeout=timeout).decode()
    if not out.strip():
        return {}
    return json.loads(out)
